# Сравнение целых чисел:
# В Bash для сравнения чисел используются различные операторы. Некоторые работают только в двойных квадратных скобках ([[ ]]), а другие — в двойных круглых скобках (( )).

# Операторы для сравнения:
[[ -eq ]]  [[ = ]]   (( = ))   # равно  
[[ -ne ]]  [[ != ]]  (( != ))  # не равно  
[[ -gt ]]  [[ > ]]   (( > ))   # больше  
[[ -ge ]]  [[ >= ]]  (( >= ))  # больше или равно  
[[ -lt ]]  [[ < ]]   (( < ))   # меньше  
[[ -le ]]  [[ <= ]]  (( <= ))  # меньше или равно  

# Если использовать нечисловые значения с [[ ]], то будет сравниваться значение ASCII символов.

# Проверка строк:
[[ -z ]]  # строка/аргумент пустая  
[[ -n ]]  # строка/аргумент не пустая

# Логические операторы для команд:
[[ command -a command ]]  # аналогично логическому И (&&)  
[[ command -o command ]]  # аналогично логическому ИЛИ (||)

# Проверка файлов:
[[ -e file ]]  # файл существует  
[[ -f file ]]  # файл существует и является обычным файлом  
[[ -d file ]]  # файл существует и является каталогом  
[[ -r file ]]  # файл существует с разрешениями на чтение  
[[ -w file ]]  # файл существует с разрешениями на запись  
[[ -x file ]]  # файл существует с разрешениями на выполнение  
[[ -s file ]]  # файл существует и его размер больше нуля  
[[ -O file ]]  # файл существует, и его владельцем является текущий пользователь  
[[ -G file ]]  # файл существует, и его владельцем является группа текущего пользователя  
[[ -h file ]]  # файл существует и является символической ссылкой  
[[ -L file ]]  # файл существует и является символической ссылкой  
[[ -b file ]]  # файл существует и является блочным устройством  
[[ -c file ]]  # файл существует и является специальным символьным файлом  
[[ -S file ]]  # файл существует и является сокетом  
[[ -N file ]]  # файл существует и был изменен с момента последнего чтения  
[[ file1 -ef file2 ]]  # файлы file1 и file2 принадлежат одному и тому же иноду  
[[ file1 -nt file2 ]]  # файл file1 новее, чем файл file2  
[[ file1 -ot file2 ]]  # файл file1 старше, чем файл file2

# Арифметические операторы:
id++  id--        # постинкремент и постдекремент  
++id  --id        # преинкремент и предекремент  
- +               # унарный минус и плюс  
! ~               # логическое и побитовое отрицание  
**                # возведение в степень  
* / %             # умножение, деление, остаток  
+ -               # сложение, вычитание  
<< >>             # побитовые сдвиги влево и вправо  
== !=             # равенство и неравенство  
&                 # побитовое И  
^                 # побитовое исключающее ИЛИ  
|                 # побитовое ИЛИ  
&&                # логическое И  
||                # логическое ИЛИ  
expr ? expr : expr  # условный оператор  
= *= /= %= += -= <<= >>= &= ^= |=  # операторы присваивания  
expr1 , expr2       # запятая
